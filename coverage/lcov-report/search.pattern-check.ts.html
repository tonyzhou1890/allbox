/**
 * @name patternCheck
 * @param {SearchPatternCfg} cfg
 * @param {string} text input string
 */
function patternCheck (cfg: SearchPatternCfg, text: string): {
  result: boolean,
  values: { field?: SearchDecorator, fieldName: string, decorator?: SearchDecorator, decoratorName: string, values: string[] }[]
  purified: string
  msg: string
} {
  const res = {
    result: true,
    values: [],
    purified: '',
    msg: ''
  }
  // 空字符串直接通过检查
  if (!text) {
    res.msg = cfg.tip ?? '输入字符或者 ' + cfg.patterns.filter(pattern => pattern.field).map(pattern => '/' + pattern.field).join('、')
    return res
  }
 
  const textArr = [...text]
  // 检测到的模式
  const patterns = []
  // 当前模式
  let pattern = {
    field: '',
    decorator: '',
    value: ''
  }
  // 解析模式
  for (let i = 0; i < textArr.length; i++) {
    const ch = textArr[i]
    // 默认的空模式只能在字符串开头
    if (!i && ch !== '/') {
      pattern.value = ch
      continue
    }
    // 遇到 ‘/’，模式开始
    if (ch === '/') {
      if (pattern.field || pattern.decorator || pattern.value) {
        patterns.push({...pattern})
      }
      pattern = {
        field: '',
        decorator: '',
        value: ''
      }
      continue
    }
    // 遇到 ‘:’，修饰符开始或者值开始，如果 pattern 里 value 不为空，说明之前已经有了一个冒号，之前的值应该是修饰符
    if (ch === ':') {
      // 如果 field 为空，说明之前的 value 介于 / 和 : 之间，应该是 field
      if (!pattern.field) {
        pattern.field = pattern.value
        pattern.value = ''
      } else if (!pattern.decorator) {
        // 如果 decorator 为空，说明之前的 value 应该是修饰符
        pattern.decorator = pattern.value
        pattern.value = ''
      } else {
        // field 和 decorator 都不为空，但依然遇到了冒号，则此冒号不处理，视为 value 的一部分
        pattern.value += ch
      }
      continue
    }
    pattern.value += ch
  }
  if (pattern.value) {
    patterns.push({...pattern})
  }
  console.log(patterns)
  // 校验模式
  return res
}
 
export default patternCheck
 
export interface SearchDecorator {
  name: string
  values?: string[]
  seperator?: string
  multiple?: boolean
  tip?: string
  /**
   * @desc 修饰符对值的校验优先于模式校验，如果配置了这个函数，将不会调用模式的校验。
   */
  validate?: (values: string[]) => { result: boolean, msg: string }
}
 
export interface SearchPattern {
  /**
   * @desc 如果 field 为空，表示空模式（字段），
   * 如果不需要对空模式进行行为控制（decorator），
   * 可以不声明空模式，模式检查会将第一个斜杠 / 之前的字符串视为空模式
   */
  field?: string
  decorators?: SearchDecorator[]
  tip?: string
  validate?: (values: string[], decorator?: SearchDecorator) => { result: boolean, msg: string }
}
 
export interface SearchPatternCfg {
  /**
   * @desc 模式数量限制，默认一个，0 表示无限制
   */
  max?: number
  tip?: string
  patterns: SearchPattern[]
  validate?: (values: {
    pattern: SearchPattern
    decorator: SearchDecorator
    values: string[]
  }) => { result: boolean, msg: string }
}
